Get started

Start a new project with Medians PS

After downloading the file, you should go to the script through the URL and follow the instructions,

Also make sure you have created the Database at your server,

Some fields you should prepare to start

  • Following the steps at the installer :
    • - Database Host
    • - Database name
    • - Database username
    • - Database password ( optional )
  • Next :
    • - Master user name
    • - Master user email
    • - Master user password
    • - Default system name

Now go to your installer http://domain.com/installer/index.php

/*
|--------------------------------------------------------------------------
| Component example
|--------------------------------------------------------------------------
|
| Import datatable actions to append as extra column in data-table rows.
| Only used for actions row like Edit/Delete/...
|
*/

import dataTableActions from './includes/data-table-actions.vue';

export default 
{
    components:{
        dataTableActions,
    },
    name:'Branches',
    data() {
        return {

            // Absolute url + component route + ( load As Json ) parameter
            url: this.conf.url+this.path+'?load=json',

            /**
            * This Object properties
            * Should be matched with the 
            * API response 
            */
            content: {
                title: '',
                users: [],
                items: [],
                columns: [],
            },

            activeItem:{},
            showAddSide:false,
            showEditSide:false,
            showLoader: true,
        }
    },



    computed: {

        /**
        * Optional for components with Data-Table only 
        * 
        */
        bindings() {


            /**
            * Optional for appending actions to the 
            * Datatable row
            */
            this.content.columns.push({
                    key: this.__("actions"),
                    component: dataTableActions,
                    sortable: false,
                });

            return {

                columns: this.content.columns,
                data: this.content.items
            }
        }
    },


    /**
    * Default props for 
    * all dynamic Components 
    */
    props: [
        'path',
        'lang',
        'setting',
        'conf',
        'auth',
    ],
    mounted() 
    {

        /**
        * Load the data after loading
        */
        this.load()
    },

    methods: 
    {
        /**
         * Check if the user is Master
         */
        isMaster()
        {
            return (this.auth && this.auth.role_id == 1);
        } , 


        /**
        * Used for data actions  
        */
        handleAction(actionName, data) {
            switch(actionName) 
            {
                case 'view':
                    // window.open(this.conf.url+data.content.prefix)
                    break;  

                case 'edit':
                    this.showEditSide = true; 
                    this.showAddSide = false; 
                    this.activeItem = data
                    break;  

                case 'delete':
                    this.$parent.delete(data, 'Branch.delete');
                    break;  
            }
        },

        /**
        * Load the data through the API
        * Send the response to update the
        * components values
        * 
        */
        load()
        {
            this.showLoader = true;
            this.$parent.handleGetRequest( this.url ).then(response=> {
                this.setValues(response)
                this.showLoader = false;
            });
        },
        

        /**
        * Set the response to ( content ) 
        */
        setValues(data) {
            this.content = JSON.parse(JSON.stringify(data)); return this
        },

        /**
        * Used to translate the content
        * emits the App.vue function
        *
        */
        __(i)
        {
            return this.$root.$children[0].__(i);
        }
    }
};
</script>

Routes

Working with routes and connect with components

We are using https://github.com/noahbuscher/macaw as system router

To get any route or add your new route you can go to /app/config/route.php and edit as the following 

                            

/**
| This example is the root path 
| 
| @param  String for route path
| @param  Class name of the controller
| @param  String  name of the function
|
|      Ends with String  name of the function
|  
*/
Macaw::get('/', \Medians\Pages\Application\PageController::class.'@page');



/**
| This example uses /login path
| 
| path /login 
|
| @param  String for route path
| @param  Class name of the controller
|
|      Ends with String  name of the function
|  
*/
Macaw::get('/login', \Medians\Auth\Application\AuthService::class.'@loginPage');


/**
| This example uses post request 

| path /API/create
| 
| @param  String for route path
| @param  Class name of the controller
|
|      Ends with String  name of the function
|  
*/
Macaw::post('/API/create', \Medians\APIController::class.'@create');


/**
| This example uses dynamic paramater  /activate-account/sd954uosdkx4w0j451dsascuasc273r0 path
| 
| @param  String for route path { :all } tag is required parameter for the function
| @param  Class name of the controller
|
|      Ends with String  name of the function
|  
*/
Macaw::get('/activate-account/(:all)', \Medians\Users\Application\UserController::class.'@activate_account');
